### Implementation Plan
1. **Command Line Interface (CLI)**: Use a library like `argparse` to handle command-line arguments.
2. **File Reading**: Use Python's built-in `csv` module for CSV files and simple string splitting for space-delimited files.
3. **Column Extraction**: Implement logic to handle both column numbers and headers.
4. **Output**: Print the extracted columns to the console or write them to a new file.

### Step 1: Set Up the CLI
Let's start by setting up the CLI using `argparse`. This will allow us to specify the file path, column specifications, and delimiter.

```python
import argparse

def setup_cli():
    parser = argparse.ArgumentParser(description='Extract specific columns from a CSV or space-delimited text file.')
    parser.add_argument('file_path', type=str, help='Path to the input file')
    parser.add_argument('columns', type=str, help='Column numbers or headers to extract (comma-separated)')
    parser.add_argument('--delimiter', type=str, default=',', help='Delimiter for the input file (default: comma)')
    parser.add_argument('--output', type=str, default=None, help='Path to the output file (default: stdout)')
    return parser.parse_args()

args = setup_cli()
```

def extract_columns(data, columns, delimiter):

import argparse
import csv
import sys

def setup_cli():
    parser = argparse.ArgumentParser(description='Extract specific columns from a CSV or space-delimited text file.')
    parser.add_argument('file_path', type=str, help='Path to the input file')
    parser.add_argument('columns', type=str, help='Column numbers or headers to extract (comma-separated)')
    parser.add_argument('--delimiter', type=str, default=',', help='Delimiter for the input file (default: comma)')
    parser.add_argument('--output', type=str, default=None, help='Path to the output file (default: stdout)')
    return parser.parse_args()

def read_file(file_path, delimiter):
    with open(file_path, 'r', newline='') as file:
        if delimiter == ',':
            reader = csv.reader(file)
        else:
            reader = csv.reader(file, delimiter=delimiter)
        return list(reader)

def extract_columns(data, columns, delimiter):
    if delimiter == ',':
        headers = data[0]
        column_indices = []
        for col in columns.split(','):
            if col.isdigit():
                column_indices.append(int(col) - 1)
            else:
                column_indices.append(headers.index(col))
        extracted_data = [[row[i] for i in column_indices] for row in data]
    else:
        column_indices = [int(col) - 1 for col in columns.split(',')]
        extracted_data = [[row[i] for i in column_indices] for row in data]
    return extracted_data

def output_data(extracted_data, output_path, delimiter):
    if output_path:
        with open(output_path, 'w', newline='') as file:
            writer = csv.writer(file, delimiter=delimiter)
            writer.writerows(extracted_data)
    else:
        writer = csv.writer(sys.stdout, delimiter=delimiter)
        writer.writerows(extracted_data)

if __name__ == '__main__':
    args = setup_cli()
    data = read_file(args.file_path, args.delimiter)
    extracted_data = extract_columns(data, args.columns, args.delimiter)
    output_data(extracted_data, args.output, args.delimiter)
